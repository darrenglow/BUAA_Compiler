# ADK 2023 sysy to mips compiler
.data 0x10008000
c: .word 1,2,3,4,
str_0: .asciiz "y is "
str_1: .asciiz "\n"
.text

# ===============main=================
Func_main:
LABEL_3:
addiu $a2 $sp -16
li $a3 5
sw $a3 0($a2)
addiu $t0 $sp -12
li $t1 6
sw $t1 0($t0)
addiu $t2 $sp -8
li $t3 7
sw $t3 0($t2)
addiu $t4 $sp -4
li $t5 8
sw $t5 0($t4)
# clear Register
# save to value T6
sw $t4 -32($sp)
# save to value T5
sw $t2 -28($sp)
# save to value T4
sw $t0 -24($sp)
# save to value T3
sw $a2 -20($sp)
# end clear Register
sw $ra 0($sp)
addiu $a0 $sp -44
addiu $t6 $gp 8
sw $t6 -4($a0)
move $sp $a0
jal Func_func
addiu $sp $sp 44
lw $ra 0($sp)
move $t7 $v0
move $s0 $t7
li $a0 268468240
li $v0 4
syscall
move $a0 $s0
li $v0 1
syscall
li $a0 268468246
li $v0 4
syscall
li $v0 10
syscall

# ===============func2=================
Func_func2:
LABEL_1:
lw $s2 -4($sp)
addiu $s1 $s2 4
lw $s3 0($s1)
move $v0 $s3
jr $ra

# ===============func=================
Func_func:
LABEL_2:
li $s4 1
# clear Register
# save to value a
sw $s4 -8($sp)
# save to value T1
sw $s3 -8($sp)
# save to value d
sw $s2 -4($sp)
# save to value T0
sw $s1 -4($sp)
# save to value y
sw $s0 -20($sp)
# save to value T7
sw $t7 -36($sp)
# save to value P_c
sw $t6 -8($sp)
# save to value T6
sw $t4 -32($sp)
# save to value T5
sw $t2 -28($sp)
# save to value T4
sw $t0 -24($sp)
# save to value T3
sw $a2 -20($sp)
# end clear Register
sw $ra 0($sp)
addiu $a0 $sp -16
addiu $s5 $sp -4
sw $s5 -4($a0)
move $sp $a0
jal Func_func2
addiu $sp $sp 16
lw $ra 0($sp)
move $s6 $v0
move $v0 $s6
jr $ra
